Backbone Mixin

Update this view every time a backbone model updates:


    var Comment = React.createClass({
        mixins: [BackboneMixin],
        getBackboneModels: function() {
            return [this.props.model];
        },
        render: function() { ... }
    });

Button Group

A nice looking replacement for a related set of buttons (like radio buttons).


    var Options = React.createClass({
        render: function() {
            return <div>
                Options
                <ButtonGroup value='1'
                             buttons={[
                                {value: '1', text: '1'},
                                {value: '2', text: '2'},
                                {value: '3', text: '3'}
                             ]}
                             onChange={() => {}} />
            </div>;
        }
    });

$_

A pseudo-component useful for internationalization.


    var translated = <$_ first="Motoko" last="Kusanagi">
        Hello, %(first)s %(last)s!
    </$_>;

Layered Component Mixin

Render another component in a div appended to the page


    var ButtonWithDialog = React.createClass({
        render: function() {
            return <button onClick={() => this.setState({clicked: true})}>
                Click Me!
            </button>;
        },
        renderLayer: function() {
            if (!this.state.clicked) {
                return null;
            }

            return <Dialog>
                clicked!
            </Dialog>;
        }
        ...
    });

Set Interval Mixin

Like setInterval that calls clearInterval for you.


    var RefreshingComponent = React.createClass({
        mixins: [SetIntervalMixin],
        componentDidMount: function() {
            this.setInterval(this.checkForUpdates, 30*1000);
        }
    });

TimeAgo

Human friendly display of how long it's been since an event happened.


    var Comment = React.createClass({
        render: function() {
            return <div>
                <FormatCommentBody body={this.props.body} />

                <div>
                    <UserBadge user={this.props.user} /> -
                    <TimeAgo time={this.props.date} />
                </div>
            </div>;
        }
    });
